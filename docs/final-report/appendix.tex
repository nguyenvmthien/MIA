\chapter{Hướng dẫn Cài đặt và Chạy}

\section*{Yêu cầu Hệ thống}

\begin{itemize}
  \item Python 3.11+
  \item Docker và Docker Compose
  \item Ollama (cài tại \url{https://ollama.ai})
  \item HuggingFace token (chấp nhận điều khoản Pyannote)
  \item RAM: $\geq$8GB (16GB recommended)
  \item GPU (optional): NVIDIA với CUDA $\geq$11.8 cho WhisperX large-v3
\end{itemize}

\section*{Cài đặt và Chạy}

\begin{lstlisting}[language=bash, caption={Các bước cài đặt và khởi động hệ thống.}]
# 1. Clone repo
git clone <repo-url> && cd MEETING-AGENT

# 2. Cai dependencies
pip install -e ".[dev]"

# 3. Cau hinh environment
cp .env.example .env
# Sua .env: dien HF_TOKEN, OLLAMA_BASE_URL

# 4. Pull Ollama models
ollama pull qwen2.5:3b
ollama pull nomic-embed-text

# 5. Khoi dong full stack
docker compose up -d

# 6. Kiem tra health
curl http://localhost:8000/health
# {"status": "ok"}

# 7. Mo Streamlit UI
streamlit run streamlit_app.py
# -> http://localhost:8501

# 8. API docs
# -> http://localhost:8000/docs

# 9. Grafana dashboard
# -> http://localhost:3000 (admin/admin)
\end{lstlisting}

\chapter{Ví dụ API Request và Response}

\section*{Submit Meeting}

\begin{lstlisting}[language=bash, caption={POST /meetings --- Submit file audio.}]
curl -X POST http://localhost:8000/meetings \
  -F "audio=@meeting.mp3" \
  -F 'roster_json={
    "workers": [
      {"worker_id":"w1","name":"Alice Chen","role":"PM",
       "email":"alice@example.com","aliases":["Alice"]},
      {"worker_id":"w2","name":"Bob Kim","role":"Engineer",
       "email":"bob@example.com","aliases":["Bob"]}

\chapter{Cấu hình Môi trường}

\begin{lstlisting}[language=bash, caption={File .env.example với các biến cấu hình.}]
# Database
DATABASE_URL=postgresql://meeting:meeting@localhost:5432/meeting_agent

# Redis / Celery
REDIS_URL=redis://localhost:6379/0
CELERY_BROKER_URL=redis://localhost:6379/0
CELERY_RESULT_BACKEND=redis://localhost:6379/1

# Ollama
OLLAMA_BASE_URL=http://localhost:11434
OLLAMA_LLM_MODEL=qwen2.5:3b
OLLAMA_EMBED_MODEL=nomic-embed-text

# Multi-Ollama (optional)
# OLLAMA_ENDPOINTS=http://host1:11434,http://host2:11434

# WhisperX
WHISPER_MODEL=base         # large-v3 for production
WHISPER_DEVICE=cpu         # cuda for GPU
WHISPER_COMPUTE_TYPE=int8
WHISPER_LANGUAGE=en

# Pyannote (required for diarization)
HF_TOKEN=hf_your_token_here
ENABLE_DIARIZATION=true

# LangSmith (optional tracing)
LANGCHAIN_TRACING_V2=false
LANGCHAIN_API_KEY=
LANGCHAIN_PROJECT=meeting-agent

# Storage
AUDIO_STORAGE_PATH=./data/audio
TRANSCRIPT_STORAGE_PATH=./data/transcripts
WORKERS_STORAGE_PATH=./data/workers.json

# Retraining
RETRAIN_MIN_CORRECTIONS=50
RETRAIN_OUTPUT_DIR=models/qwen-meeting-latest
MLFLOW_TRACKING_URI=http://localhost:5000

# App
LOG_LEVEL=INFO
MAX_AUDIO_DURATION_HOURS=4
LLM_MAX_RETRIES=3
TASK_CONFIDENCE_THRESHOLD=0.6
