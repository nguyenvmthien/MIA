\chapter{Hướng dẫn chạy}

\section{Các lệnh chính}

\begin{lstlisting}[language=bash,caption={Các lệnh thường dùng}]
docker compose up -d --build
make test
make hygiene
make train-baseline
make benchmark
make mlops-up
\end{lstlisting}

\section{URL local quan trọng}

\begin{itemize}[leftmargin=*]
  \item Frontend: \texttt{http://localhost:3001}
  \item API health: \texttt{http://localhost:8000/health}
  \item Prometheus metrics: \texttt{http://localhost:8000/metrics}
  \item Prometheus UI: \texttt{http://localhost:9090}
  \item Grafana: \texttt{http://localhost:3000}
\end{itemize}

\section{Biến môi trường chính}

\footnotesize
\begin{longtable}{@{}>{\raggedright\arraybackslash}p{5.2cm}p{9.3cm}@{}}
\toprule
\textbf{Biến} & \textbf{Mục đích} \\
\midrule
\codefield{CONFIG_FILE} & Đường dẫn file YAML runtime. \\
\codefield{DATABASE_URL} & Chuỗi kết nối PostgreSQL. \\
\codefield{REDIS_URL} & Redis broker/cache URL. \\
\codefield{OLLAMA_BASE_URL} & Đường dẫn Ollama. \\
\codefield{OLLAMA_LLM_MODEL} & Tên LLM, mặc định \texttt{qwen2.5:3b}. \\
\codefield{WHISPER_MODEL} & Kích thước model WhisperX. \\
\codefield{HF_TOKEN} & Token cho diarization model trong cấu hình có bật speaker diarization. \\
\codefield{RETRAIN_MIN_CORRECTIONS} & Ngưỡng feedback để kiểm tra retraining. \\
\codefield{MLFLOW_TRACKING_URI} & Đường dẫn MLflow tracking. \\
\bottomrule
\end{longtable}

Secret được đặt trong \texttt{.env}; repo chỉ giữ cấu hình mẫu không nhạy cảm.
