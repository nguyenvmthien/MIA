% ── The Problem ────────────────────────────────────────────────────────────
\begin{frame}{The Problem}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{block}{Real pain points}
        \begin{itemize}
          \item After a meeting, nobody is fully sure who should do what.
          \item Manual note-taking takes 30--60 minutes.
          \item Sending audio to cloud services risks exposing internal data.
        \end{itemize}
      \end{block}
      \begin{block}{Design question}
        Can we automate this while keeping all data local?
      \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{tabular}{@{}lp{4cm}@{}}
        \toprule
        \textbf{Problem} & \textbf{Consequence} \\
        \midrule
        No minutes & Tasks are missed \\
        Unclear owner & Work drifts \\
        Cloud upload & Data exposure \\
        \bottomrule
      \end{tabular}
    \end{column}
  \end{columns}
\end{frame}

% ── Vision ─────────────────────────────────────────────────────────────────
\begin{frame}{Vision}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.44\textwidth}
      \begin{block}{Goal}
        Upload audio $\rightarrow$ receive a summary and tasks with assignee, deadline, and priority.
      \end{block}
      \begin{block}{Key principle}
        If the system is not sure, it flags the task for human review instead of guessing.
      \end{block}
    \end{column}
    \begin{column}{0.52\textwidth}
      \begin{block}{Differentiators}
        \begin{tabular}{@{}lccc@{}}
          \toprule
           & ChatGPT & Otter.ai & \textbf{Ours} \\
          \midrule
          100\% local & \texttimes & \texttimes & \checkmark \\
          Auto-assign & \texttimes & \texttimes & \checkmark \\
          Fine-tunable & \texttimes & \texttimes & \checkmark \\
          Self-learning & \texttimes & \texttimes & \checkmark \\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ── Evaluation Results ─────────────────────────────────────────────────────
\begin{frame}{Evaluation Results}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\textwidth}
      \begin{block}{Results on 5-sample gold set (13 labeled tasks)}
        \begin{tabular}{@{}lcc@{}}
          \toprule
          Metric & Result & Target \\
          \midrule
          Precision        & 0.767  & $\geq$ 0.85 \\
          Recall           & 0.700  & $\geq$ 0.90 \\
          F1               & 0.727  & ---         \\
          Schema failure   & 0.0\%  & $\leq$ 5\%  \\
          Hallucination    & 0      & $\leq$ 5\%  \\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{What it means}
        \begin{itemize}
          \item Guardrails are working correctly.
          \item Precision/Recall gap is mainly due to paraphrase mismatch (Jaccard $<$ 0.6).
          \item Fine-tuning with 50+ samples is expected to reach Precision $\approx$ 0.87.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
